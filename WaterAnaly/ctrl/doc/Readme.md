```text
const 方向_吸液 = "FORWARD"   // 正转：吸液
const 方向_推液 = "REVERSE"   // 反转：推液/吹气

// 采样与去抖
const 采样周期_ms = 20
const 稳定次数 = 10

// 阀切换稳定
const 阀切换稳定_ms = 50

// 超时
const 超时_取大_ms = 15000
const 超时_取小_ms = 15000
const 超时_排空_ms = 10000
const 超时_抽消解_ms = 12000
const 升温超时_ms = 180000

// 排空补吹
const 补吹_ms = 500

// 工艺参数（可改）
const 消解温度 = 50
const 加热保持_ms = 300000     // 5min
const 静置消解总_ms = 900000   // 15min
const 搅拌时长_ms = 15000      // 15s
const 测光预热_ms = 2000

// 阈值（标定/配置文件）
config 上满阈值
config 下满阈值
config 空阈值

// -----------------------
function sleep_ms(ms)
function now_ms() -> int

function wait_until(cond_fn, timeout_ms, poll_ms) -> bool:
    start = now_ms()
    while (not cond_fn()) and (now_ms() - start < timeout_ms):
        sleep_ms(poll_ms)
    return cond_fn()

function 稳定满足(cond_fn) -> bool:
    ok = 0
    i = 0
    while i < 稳定次数:
        if cond_fn():
            ok = ok + 1
        sleep_ms(采样周期_ms)
        i = i + 1
    return ok == 稳定次数
```

# 1) 基础原语层 primitives

```text
// =======================
// 硬件抽象（你们Linux驱动实现）
// =======================

class 阀控:
    function 打开(阀列表)
    function 关闭(阀列表)
    function 全关()

class 蠕动泵:
    function 设置方向(dir)   // "FORWARD"/"REVERSE"
    function 启用()
    function 禁用()

class 温控:
    function 启动(目标温度)
    function 停止()
    function 读温度() -> float

class 计量光学:
    function 读上透光度() -> float
    function 读下透光度() -> float

class 消解光学:
    function 开光源()
    function 关光源()
    function 读吸光度() -> float


// =======================
// 基础判定（满/空）
// =======================

function 满(计量光学obj, 量) -> bool:
    if 量 == "大":
        return 稳定满足(() => 计量光学obj.读上透光度() <= 上满阈值)
    else:
        return 稳定满足(() => 计量光学obj.读下透光度() <= 下满阈值)

function 空(计量光学obj) -> bool:
    return 稳定满足(() => 计量光学obj.读上透光度() >= 空阈值)


// =======================
// 路由（统一写法）
// =======================

function 路由_源到计量(阀, 源阀):
    阀.全关()
    阀.打开([源阀, "计量单元入口"])
    sleep_ms(阀切换稳定_ms)

function 路由_计量到目标(阀, 目标阀列表):
    阀.全关()
    阀.打开(目标阀列表)
    sleep_ms(阀切换稳定_ms)


// =======================
// 核心动作：取 / 打（打包含排空+补吹）
// =======================

function 取(阀, 泵, 计量光学obj, 源阀, 量):
    timeout = (量 == "大") ? 超时_取大_ms : 超时_取小_ms

    路由_源到计量(阀, 源阀)

    泵.设置方向(方向_吸液)
    泵.启用()

    ok = wait_until(() => 满(计量光学obj, 量), timeout, 50)

    泵.禁用()
    阀.全关()

    if not ok:
        throw "取液超时(缺液/堵塞/气泡/阀未到位)"

function 打(阀, 泵, 计量光学obj, 目标阀列表):
    // 目标阀列表可包含多个阀，例如：["消解器下阀","消解器上阀"]
    路由_计量到目标(阀, 目标阀列表)

    泵.设置方向(方向_推液)
    泵.启用()

    ok = wait_until(() => 空(计量光学obj), 超时_排空_ms, 50)

    泵.禁用()

    if not ok:
        阀.全关()
        throw "排空失败(堵塞/废液阻塞/阀未到位)"

    // ===== 强制补吹：把目标管段余液吹尽 =====
    泵.设置方向(方向_推液)
    泵.启用()
    sleep_ms(补吹_ms)
    泵.禁用()

    // 结束状态：计量单元为空气、阀全关
    阀.全关()


// =======================
// 通气搅拌（基础动作）
// =======================

function 通气搅拌_消解器(阀, 泵, 时长_ms):
    阀.全关()
    阀.打开(["消解器上阀", "消解器下阀"])
    sleep_ms(阀切换稳定_ms)

    泵.设置方向(方向_推液)   // 反转吹气
    泵.启用()
    sleep_ms(时长_ms)
    泵.禁用()

    阀.全关()
```

# 2) 工序封装层 operations

```text
// =======================
// 工序封装：固化规则（双阀/排空/清洗）
// =======================

function 向消解器加液(阀, 泵, 计量光学obj, 源阀, 量):
    // 规则：加液必须双阀开启
    取(阀, 泵, 计量光学obj, 源阀, 量)
    打(阀, 泵, 计量光学obj, ["消解器下阀", "消解器上阀"])

function 润洗(阀, 泵, 计量光学obj, 源阀, 废液阀):
    // 规则：润洗=小量取+排废（自动排空+补吹）
    取(阀, 泵, 计量光学obj, 源阀, "小")
    打(阀, 泵, 计量光学obj, [废液阀])

function 冲洗管路(阀, 泵, 计量光学obj, 源阀, 废液阀, 次数, 量):
    i = 0
    while i < 次数:
        取(阀, 泵, 计量光学obj, 源阀, 量)
        打(阀, 泵, 计量光学obj, [废液阀])
        sleep_ms(200)
        i = i + 1

function 排空消解器(阀, 泵, 计量光学obj, 废液阀):
    // Step1: 消解器内容抽到计量单元（双阀+计量入口）
    阀.全关()
    阀.打开(["消解器下阀", "消解器上阀", "计量单元入口"])
    sleep_ms(阀切换稳定_ms)

    泵.设置方向(方向_吸液)
    泵.启用()

    ok = wait_until(() => 满(计量光学obj, "大"), 超时_抽消解_ms, 50)

    泵.禁用()
    阀.全关()

    if not ok:
        throw "抽取消解器超时(消解器无液/堵塞/阀未到位)"

    // Step2: 计量 -> 废液（自动排空+补吹）
    打(阀, 泵, 计量光学obj, [废液阀])

function 全面清洗(阀, 泵, 计量光学obj):
    // 1) 清洗消解器
    向消解器加液(阀, 泵, 计量光学obj, "标一", "大")
    sleep_ms(20000)
    排空消解器(阀, 泵, 计量光学obj, "废液1")

    // 2) 清洗计量/中枢管路
    冲洗管路(阀, 泵, 计量光学obj, "标一", "废液1", 3, "大")

function 升温并恒温(温控obj, 目标温度, 恒温_ms):
    温控obj.启动(目标温度)

    ok = wait_until(() => 温控obj.读温度() >= 目标温度, 升温超时_ms, 500)

    if not ok:
        温控obj.停止()
        throw "升温超时(加热异常/温度传感器异常)"

    sleep_ms(恒温_ms)
    温控obj.停止()

function 读取吸光度(消解光学obj) -> float:
    消解光学obj.开光源()
    sleep_ms(测光预热_ms)
    A = 消解光学obj.读吸光度()
    消解光学obj.关光源()
    return A
```

# 3) 配方流程层 recipe (氨氮/水杨酸法)

```text
// =======================
// 配方流程：只写顺序与参数
// =======================

function 氨氮检测流程(阀, 泵, 计量光学obj, 温控obj, 消解光学obj, 标二浓度) -> float:
    阀.全关()

    // 0) 开机预清洗
    全面清洗(阀, 泵, 计量光学obj)

    // 1) 水样：润洗 + 正式加样
    润洗(阀, 泵, 计量光学obj, "待测溶液", "废液1")
    向消解器加液(阀, 泵, 计量光学obj, "待测溶液", "大")

    // 2) 标二：润洗 + 加标（按你现有工艺保留）
    润洗(阀, 泵, 计量光学obj, "标二", "废液1")
    向消解器加液(阀, 泵, 计量光学obj, "标二", "大")

    // 3) 试剂A：润洗 + 加液 + 用标一清洗1~2次（你原流程是2次）
    润洗(阀, 泵, 计量光学obj, "试剂A", "废液1")
    向消解器加液(阀, 泵, 计量光学obj, "试剂A", "大")
    冲洗管路(阀, 泵, 计量光学obj, "标一", "废液1", 2, "大")

    // 4) 加热 50℃ 5min
    升温并恒温(温控obj, 消解温度, 加热保持_ms)

    // 5) 试剂B：润洗 + 加液 + 标一清洗1次
    润洗(阀, 泵, 计量光学obj, "试剂B", "废液1")
    向消解器加液(阀, 泵, 计量光学obj, "试剂B", "大")
    冲洗管路(阀, 泵, 计量光学obj, "标一", "废液1", 1, "大")

    // 6) 静置消解 15min，中间通气搅拌一次（15秒）
    sleep_ms(静置消解总_ms / 2)
    通气搅拌_消解器(阀, 泵, 搅拌时长_ms)
    sleep_ms(静置消解总_ms / 2)

    // 7) 试剂C：润洗 + 加液 + 标一清洗2次
    润洗(阀, 泵, 计量光学obj, "试剂C", "废液1")
    向消解器加液(阀, 泵, 计量光学obj, "试剂C", "大")
    冲洗管路(阀, 泵, 计量光学obj, "标一", "废液1", 2, "大")

    // 8) 比色测量
    A = 读取吸光度(消解光学obj)

    // 9) 浓度计算（你们自定义）
    氨氮浓度 = 计算浓度(A, 标二浓度)
    显示结果(氨氮浓度)

    // 10) 收尾：排液 + 清洗 + 复位
    排空消解器(阀, 泵, 计量光学obj, "废液1")
    全面清洗(阀, 泵, 计量光学obj)
    阀.全关()

    return 氨氮浓度
```

## 这版“完整伪代码”解决了你之前的关键风险点

* 润洗 = 小量取 + 排废（并且**排空+补吹**）
* 每次打出后，系统回到统一状态：**计量空 / 管路空气 / 阀全关**
* 向消解器加液强制**双阀开启**
* 排空消解器走“抽到计量→打到废液”，并继承排空规则
